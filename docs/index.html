import requests, json, os from datetime import datetime def fetch_twse(date_str): url = f"https://www.twse.com.tw/rwd/zh/fund/T86?date={date_str}&selectType=ALLBUT0999&response=json" r = requests.get(url) r.encoding = "utf-8" data = r.json() result = [] if "data" in data: for row in data["data"]: stock_id = row[0].strip() name = row[1].strip() foreign = int(row[2].replace(",", "")) - int(row[3].replace(",", "")) trust = int(row[4].replace(",", "")) - int(row[5].replace(",", "")) dealer = int(row[6].replace(",", "")) - int(row[7].replace(",", "")) total = foreign + trust + dealer result.append({ "stock": stock_id, "name": name, "foreign": foreign, "trust": trust, "dealer": dealer, "total": total, "date": date_str }) return result def update_daily(): date_str = datetime.today().strftime("%Y%m%d") twse_data = fetch_twse(date_str) output_path = "docs/data.json" if os.path.exists(output_path): with open(output_path, "r", encoding="utf-8") as f: merged = json.load(f) else: merged = {} for item in twse_data: stock_id = item["stock"] merged.setdefault(stock_id, []) # 移除同一天舊資料 merged[stock_id] = [x for x in merged[stock_id] if x["date"] != date_str] merged[stock_id].append(item) # 按日期排序 & 只保留最近 20 筆 merged[stock_id] = sorted(merged[stock_id], key=lambda x: x["date"], reverse=True)[:20] with open(output_path, "w", encoding="utf-8") as f: json.dump(merged, f, ensure_ascii=False, indent=2) print(f"{date_str} 完成更新，共 {len(twse_data)} 筆 → 已更新 {output_path}") if __name__ == "__main__": update_daily()
